int main(void)
{

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_USB_HOST_Init();
  MX_ADC1_Init();
  MX_CAN1_Init();
  MX_ADC2_Init();
  /* USER CODE BEGIN 2 */

  txheader.IDE=CAN_ID_STD;
  txheader.StdId=0x100;
  txheader.RTR=CAN_RTR_DATA;
  txheader.DLC=4;

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
    /* USER CODE END WHILE */
    MX_USB_HOST_Process();

    /* USER CODE BEGIN 3 */
    HAL_ADC_Start(&hadc1);
    HAL_ADC_PollForConversion(&hadc1, 100);

    HAL_ADC_Start(&hadc2);
    HAL_ADC_PollForConversion(&hadc2, 100);

    sconfig.Channel=ADC_CHANNEL_1;
    HAL_ADC_ConfigChannel(&hadc1, &sconfig);
    uint32_t fuel_level_adc = HAL_ADC_GetValue(&hadc1);

    float fuel_level =fuel_level_adc * 0.039 ;
    HAL_Delay(1000);

    sconfig2.Channel=ADC_CHANNEL_2;
    HAL_ADC_ConfigChannel(&hadc2, &sconfig2);

    uint32_t temp_level_adc = HAL_ADC_GetValue(&hadc2);

    float temp_level = temp_level_adc * 0.097 ;

    HAL_ADC_Stop(&hadc1);
    HAL_ADC_Stop(&hadc2);
    uint8_t door_lock_status =    HAL_GPIO_ReadPin(Door_Status_GPIO_Port,Door_Status_Pin);

    uint8_t seat_belt_status =    HAL_GPIO_ReadPin(Seat_Belt_Status_GPIO_Port,Seat_Belt_Status_Pin);

    txdata[0]=fuel_level;
    txdata[1]=temp_level;
    txdata[2]=door_lock_status;
    txdata[3]=seat_belt_status;

    HAL_CAN_AddTxMessage(&hcan1, &txheader, txdata, &txmailbox);

    HAL_Delay(1000);
  }
  /* USER CODE END 3 */
}
